create database auditing;
use auditing;

create table register(username varchar(50) primary key,
password varchar(50),
contact_no varchar(20),
email varchar(50), 
address varchar(80));

create table files(username varchar(50), file_name varchar(80), upload_date varchar(30), file_status varchar(30), duplicate_file varchar(60));

create table tpa(username varchar(50), file_name varchar(80), file_integrity varchar(300));
